using System.Collections;
using System.Collections.Generic;
using UnityEngine;

public class LevelRecorce : MonoBehaviour
{
    [SerializeField] private int Initial_RES_count;
    [SerializeField] private int Max_RES_count;
    [SerializeField] private float N_of_Seconds_before_next_addition;
    [SerializeField] private int N_of_Increce;
    private int RES_count;
    private float NoSb;
    void Start()
    {
        RES_count = 0;
        N_of_Seconds_before_next_addition = NoSb;
    }

    // Update is called once per frame
    void Update()
    {
        if(NoSb > 0)
        {
            NoSb -= Time.deltaTime;
        }
        else
        {
            RES_count += N_of_Increce;
        }
    }
}
